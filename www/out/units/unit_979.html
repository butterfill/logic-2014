Revison: Proofs (with quantifiers)

\begin{enumerate}
\item
\begin{equation*}
\begin{fitch}
\fh ∀x S(x) \\
\fh ∀x \lnot S(x) \\
\fa ⊥ \\
\end{fitch}
\end{equation*}
\item
\begin{equation*}
\begin{fitch}
\fh ∀x ( F(x) → x=a ) \\
\fa ¬∃x ( F(x) ∧ ¬x=a ) \\
\end{fitch}
\end{equation*}
\item
\begin{equation*}
\begin{fitch}
\fh ∃x ∀y ( F(y) → ¬G(x,y) ) \\
\fa ∀y ∃x ( F(y) → ¬G(x,y) ) \\
\end{fitch}
\end{equation*}
\end{enumerate}
